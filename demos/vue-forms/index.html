// Vue 3 Composition API — Form with v-model

const { createApp, ref, computed } = Vue

createApp({
  setup() {
    // Reactive form state — each field bound via v-model
    const form = ref({
      title: '',
      description: '',
      category: '',
      rating: ''
    })

    const attempted = ref(false)
    const saving = ref(false)
    const items = ref([])

    // Computed validation — reactive, re-evaluates on every change
    const titleError = computed(() => {
      if (!form.value.title.trim()) return 'Title is required'
      if (form.value.title.trim().length < 3) return 'Min 3 characters'
      return ''
    })

    const ratingError = computed(() => {
      if (!form.value.rating) return 'Priority is required'
      return ''
    })

    // Computed isValid — bound to :disabled on submit button
    const isValid = computed(() => {
      return !titleError.value && !ratingError.value
    })

    // Submit handler — "Saving..." for 300ms, then reset
    async function handleSubmit() {
      attempted.value = true
      if (!isValid.value) return

      saving.value = true
      await new Promise(r => setTimeout(r, 300))

      items.value.unshift({ ...form.value, id: Date.now() })
      form.value = { title: '', description: '', category: '', rating: '' }
      attempted.value = false
      saving.value = false
    }

    return { form, attempted, saving, items,
             titleError, ratingError, isValid, handleSubmit }
  }
}).mount('#app')
